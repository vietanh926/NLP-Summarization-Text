Triển khai ý tưởng trên ra code ta có như sau

#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int n, k;

ll power(int n, int k){
    ll res = 1;
    while(k--) res *= n;
    return res;
}

int main(){
    freopen("CTDL.inp","r",stdin);
    freopen("CTDL.out","w",stdout);
    cin >> n >> k;
    cout << power(n, k) << endl;

    return 0;
}

Lời giải cải tiến
Từ công thức trên, ta có thể triển khai thành code như sau

#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int n, k;

ll power(int n, int k){
    if(k == 0) return 1;
    ll res = power(n, k / 2);
    if(k % 2) return res * res * n;
    return res * res;
}

int main(){
    freopen("CTDL.inp","r",stdin);
    freopen("CTDL.out","w",stdout);
    cin >> n >> k;
    cout << power(n, k) << endl;

    return 0;
}
